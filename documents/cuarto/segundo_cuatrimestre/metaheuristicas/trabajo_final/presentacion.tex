\section{Fundamentos matemáticos}
% ===============================================================

\begin{frame}{Ecuación de actualización}
  \begin{columns}[T, onlytextwidth]
    \column{0.5\textwidth}
      \begin{block}{Regla de movimiento}
        \vspace{2pt}
        \[
          X_j^{t+1} =
          \begin{cases}
            X_j^t + r_1 \sin(r_2) |r_3 P_j^t - X_j^t|, & r_4 < \tfrac{1}{2}\\[4pt]
            X_j^t + r_1 \cos(r_2) |r_3 P_j^t - X_j^t|, & r_4 \geq \tfrac{1}{2}
          \end{cases}
        \]
        \vspace{2pt}
      \end{block}
      \vspace{0.4em}
      Definiendo $\phi = r_1 \cdot f(r_2)$ y $|\Delta_j^t| = |r_3 P_j^t - X_j^t|$:
      \vspace{0.3em}
      \begin{block}{Notación abreviada}
        \centering
        $\displaystyle X_j^{t+1} = X_j^t + \phi \cdot |\Delta_j^t|$
      \end{block}

    \column{0.47\textwidth}
      \vspace{0.3em}
      {\small
      \begin{tabular}{c p{5.1cm}}
        \toprule
        \textbf{Param.} & \textbf{Rol} \\
        \midrule
        $r_1$ & {Amplitud}: exploración/explotación \\
        $r_2$ & Ángulo: dirección y signo del paso \\
        $r_3$ & Peso de la distancia al destino \\
        $r_4$ & Selección equiprobable sin/cos \\
        \bottomrule
      \end{tabular}}
      \vspace{0.5em}

      \begin{exampleblock}{Separación magnitud/dirección}
        \centering
        $|\Delta_j^t|$ fija la magnitud $(\geq 0)$\\
        $\phi$ fija signo y escala $\Rightarrow$ simetría
      \end{exampleblock}
  \end{columns}
\end{frame}

\section{Exploración vs.\ Explotación}
% ===============================================================



% --- 3.1  Mapa de componentes -----------------------------------
\begin{frame}{¿Qué componente provoca qué?}
    \vspace{-0.5cm}
  \begin{columns}[T, onlytextwidth]
    \column{0.48\textwidth}
      \begin{exampleblock}{Componentes exploratorios}
        \begin{itemize}
          \setlength\itemsep{0.35em}
          \item \textbf{$r_1 > 1$}: posibilita $\phi\notin[-1,1]$ haciendo que el agente sobrepase el destino
          \item \textbf{$r_3 > 1$}: aleja el destino efectivo aumentando $|\Delta_j^t|$
          \item \textbf{$r_2 \in [0,2\pi]$}: cubre todas las direcciones, incluidas las de signo negativo
          \item \textbf{$r_4$ aleatorio}: seno y coseno desfasados $\pi/2$ mejoran la isotropía
        \end{itemize}
      \end{exampleblock}

    \column{0.48\textwidth}
      \begin{alertblock}{Componentes explotadores}
        \begin{itemize}
          \setlength\itemsep{0.35em}
          \item \textbf{$r_1 < 1$}: asegura $\phi\in(-1,1)$ por lo que el agente no sobrepasa $P_j^t$
          \item \textbf{$r_3 < 1$}: acerca el destino efectivo reduciendo $|\Delta_j^t|$
          \item \textbf{Destino global $P_j^t$}: único atractor compartido que concentra la población
          \item \textbf{Decrecimiento de $r_1$}: garantiza convergencia progresiva
        \end{itemize}
      \end{alertblock}
  \end{columns}

    {\small
    $r_1$ es el único parámetro \textbf{determinista}.\\
    $r_2, r_3, r_4$ son aleatorios $\Rightarrow$ el balance es \textbf{aleatorio pero dirigido}}
\end{frame}

      \begin{block}{$r_3 \in [0,2]$: destino efectivo}
        \begin{equation*}%
            |\Delta_j^t| = |{\color{myred}r_3} P_j^t - X_j^t|%
        \end{equation*}%
        \vspace{-0.5cm}
        \begin{itemize}%
          \setlength\itemsep{0.3em}%
          \item $r_3 > 1$ $\Rightarrow$ aumenta $|\Delta_j^t|$\\
                \hspace{1.5em}\hg{efecto exploratorio secundario}
          \item $r_3 < 1$ $\Rightarrow$ reduce $|\Delta_j^t|$\\
                \hspace{1.5em}\hy{efecto explotador secundario}
          \item $r_3 = 1$ $\Rightarrow$ distancia real
        \end{itemize}
        \vspace{0.2em}
        \textbf{Utilidad:} introduce diversidad {incluso cuando} $r_1 < 1$
      \end{block}

    \column{0.48\textwidth}
      \begin{block}{$r_4 \in [0,1]$: selección sin/cos}
        Probabilidad $\tfrac{1}{2}$ para cada función,
        desfasadas $\tfrac{\pi}{2}$ rad $\Rightarrow$ cobertura más uniforme

\section{Implementación}
% ===============================================================

\begin{frame}[fragile]{Pseudocódigo del SCA}
\begin{columns}[T, onlytextwidth]
    \column{0.65\textwidth}
    \vspace{-0.7cm}
\begin{lstlisting}[style=aimaStyle, language=pseudo]
(* Parámetros: N (población), T (iteraciones máx.), a=2 *)
Inicializar X = {X_1,...,X_N} aleatoriamente en [lb,ub]^d
Evaluar f(X_i) para cada i;   P <- argmin f(X_i); t <- 1
Mientras t <= T hacer
    r1 <- 2*(1 - t/T)                      (* Ec. decrecimiento *)
    Para cada agente i = 1 hasta N hacer
        Para cada dimensión j = 1 hasta d hacer
            r2 <- U[0,2pi];  r3 <- U[0,2];  r4 <- U[0,1]
            Si r4 < 0.5 entonces
                X_i[j] <- X_i[j] + r1*sin(r2)*|r3*P[j]-X_i[j]|
            Si_no
                X_i[j] <- X_i[j] + r1*cos(r2)*|r3*P[j]-X_i[j]|
            Fin Si
            X_i[j] <- max(lb[j], min(ub[j], X_i[j]))  (* clamping *)
        Fin Para
        Si f(X_i) < f(P) entonces  P <- X_i  Fin Si
    Fin Para
    t <- t + 1
Fin Mientras
Retornar P       (* mejor solución encontrada *)
\end{lstlisting}
    \column{0.35\textwidth}
    \begin{block}{Complejidad}
        \centering
        $\mathcal{O}(N \cdot d)$ por iteración\\
        $N \times T$ evaluaciones totales
    \end{block}
  \end{columns}
\end{frame}

\section{Ventajas, limitaciones y aplicaciones}
% ===============================================================

\begin{frame}{Ventajas e inconvenientes}
  \begin{columns}[T, onlytextwidth]
    \column{0.49\textwidth}
      \begin{exampleblock}{{Ventajas}}
        \begin{itemize}
          \setlength\itemsep{0.35em}
          \item Implementación simple: los únicos parámetros son $a$, $N$, $T$
          \item Balance explor./explot.\ \textbf{automático}
          \item Competitivo frente a PSO, GA, GWO
          \item Bajo coste: $\mathcal{O}(N \cdot d)$ por iteración
          \item Fácilmente hibridable (PSO, DE, ABC, \ldots)
        \end{itemize}
      \end{exampleblock}

    \column{0.49\textwidth}
      \begin{alertblock}{{Inconvenientes}}
        \begin{itemize}
          \setlength\itemsep{0.35em}
          \item Convergencia prematura en funciones multimodales
          \item Decaimiento lineal de $r_1$: no se adapta al paisaje
          \item Baja diversidad en iteraciones avanzadas
          \item Escasa memoria colectiva (solo $P_j^t$ compartido)
          \item Pobre en espacios discretos sin adaptación previa
        \end{itemize}
      \end{alertblock}
  \end{columns}

  \vspace{0.5em}
  \begin{block}{Cuándo funciona bien}
    Problemas \hl{continuos} de dimensión baja/media ($d \leq 100$) \textbullet\
    Funciones \hl{unimodales} o con pocos óptimos locales \textbullet\
    Cuando evaluar $f$ es costoso (bajo overhead)
  \end{block}
\end{frame}

% ===============================================================
\section{Conclusiones}
% ===============================================================

\begin{frame}{Conclusiones}
  \begin{enumerate}
    \setlength\itemsep{0.42em}
    \item El SCA es un algoritmo basado en matemáticas: $\sin$ y $\cos$ proveen exploración e isotropía de forma natural
    \item \hl{$r_1$} es el \textbf{director del equilibrio}: su decrecimiento convierte la búsqueda en un proceso de \textbf{enfriamiento implícito}
    \item El \textbf{umbral efectivo} de transición es $r_1^* = \pi/2$ lo que provoca que la explotación predomine desde $t \approx 0.21\,T$
    \item \hl{$r_3$} modula secundariamente: introduce \textbf{diversidad} incluso durante la explotación.
    \item Sus limitaciones (convergencia prematura, escasa memoria colectiva) han motivado una familia rica de \textbf{variantes híbridas}
    \item Como dicta el teorema \textit{No Free Lunch}: competitivo en funciones continuas suaves, pero no universalmente superior
  \end{enumerate}
\end{frame}
