\newcommand{\@smarttilde}[2]{%
  \sbox0{$#1#2$}%
  \dimen0=\wd0%
  \ifdim\dimen0<1.6ex
    \mathaccent"0365{#2}% tilde pequeña estándar
  \else\ifdim\dimen0<3.5ex
    \widetilde{#2}% tilde normal
  \else
    % tilde escalada y MÁS PEGADA
    \ThisStyle{%
      \stackengine{0.05ex}{#2}{\stretchto{\widetilde{\phantom{#2}}}{1.07\ht\strutbox}}{O}{c}{F}{F}{L}%
    }%
  \fi\fi
}
