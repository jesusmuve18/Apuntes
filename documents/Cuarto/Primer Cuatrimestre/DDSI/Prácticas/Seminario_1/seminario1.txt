# Creaci√≥n de tablas
CREATE TABLE Stock(
	Cproducto varchar2(10) CONSTRAINT Cproducto_clave_primaria PRIMARY KEY,
	Cantidad int
);

CREATE TABLE Pedido(
	Cpedido varchar2(10) CONSTRAINT Cpedido_clave_primaria PRIMARY KEY,
	Ccliente varchar2(10),
	Fecha-pedido date
);

CREATE TABLE Detalle-Pedido(
	Cpedido CONSTRAINT Cpedido_clave_externa_pedido REFERENCES Pedido(Cpedido),
	Cproducto CONSTRAINT Cproducto_clave_externa_stock REFERENCES Stock(Cproducto),
	Cantidad int
	CONSTRAINT clave_primaria PRIMARY_KEY (Cpedido,Cproducto)
);

# Borrado de tablas
DROP TABLE "Detalle-Pedido" CASCADE CONSTRAINTS PURGE;
DROP TABLE Pedido CASCADE CONSTRAINTS PURGE;
DROP TABLE Stock CASCADE CONSTRAINTS PURGE;
