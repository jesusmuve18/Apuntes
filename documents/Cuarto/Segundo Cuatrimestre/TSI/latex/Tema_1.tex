\begin{definicion}
    Un algoritmo $A$ es una secuencia ordenada, finita y bien definida de pasos o instrucciones lĂłgicas que permiten resolver un problema. 
\end{definicion}

\begin{definicion}
    Diremos que un algoritmo $A$ es \textbf{completo} si encuentra una soluciĂłn al problema en caso de que exista en un tiempo finito.
\end{definicion}

\begin{definicion}
    Diremos que un algoritmo $A$ es \textbf{admisible} si, en caso de encontrar soluciĂłn a un problema, esta es Ăłptima.
\end{definicion}

\begin{definicion}
    Dados dos algoritmos $A_1$ y $A_2$, diremos que $A_2$ \textbf{domina} a $A_1$ si todo nodo $n$ extendido por $A_1$ es tambiĂŠn expandido por $A_2$. En el caso de que $A_2$ domine tendremos que serĂĄ mĂĄs eficiente en el tiempo de bĂşsqueda.
\end{definicion}

\begin{notacion}
    A continuaciĂłn describiremos la notaciĂłn que usaremos a lo largo de estos apuntes:
    \begin{center}
        \begin{tabular}{c | l}
            $n$ & \textbf{Nodo} cualquiera.\\
            $s$ & \textbf{Estado inicial}. \\
            $\gamma$ & Nodo \textbf{objetivo}.\\
            $\Gamma$ & \textbf{Conjunto de objetivos}.\\
            $\Gamma^*$ & Conjunto de nodos a los que se puede llegar con un camino Ăłptimo.\\
            $P_{n_i-n_j}$ & \textbf{Camino arbitrario} del nodo $n_i$ al nodo $n_j$.\\
            $P^*_{n_i-n_j}$ & \textbf{Camino Ăłptimo} del nodo $n_i$ al nodo $n_j$.\\
            $P_{n_i-\Gamma}$ & \textbf{Camino arbitrario de $n_i$ a un objetivo}.\\
            $c^*$ & \textbf{Costo Ăłptimo} de $s$ a una soluciĂłn admisible.
        \end{tabular}
    \end{center}

    \begin{center}

            $h(n)$ & \textbf{Costo estimado} del nodo $n$ al objetivo $\gamma$.\\

            $f(n)$ & \textbf{Costo total estimado} al objetivo $\gamma$ pasando por el nodo $n$ ($f(n)=g(n)+h(n)$).
        \end{tabular}
    \end{center}
\end{notacion}

% Ărbol de Estados

\begin{propiedades}
    Hay dos propiedades de los ĂĄrboles de estados que estĂĄn siempre presentes:
    \begin{enumerate}
        \item El grafo es \textbf{localmente finito}, es decir, el nĂşmero de sucesores de cada nodo es finito.
        \item Se verifica que $c(n_i, n_j) \geq \delta > 0$ para cualesquiera $n_i,n_j$ nodos, es decir, el costo de cambiar de nodo es siempre positivo. 
    \end{enumerate}
\end{propiedades}

\begin{observacion}
    Un ĂĄrbol de estados puede tener varios nodos iniciales $s_1, ..., s_n$. Para simplificar el problema lo que se hace es crear un estado inicial artificial $s$ y se conecta a cada uno de los estados iniciales $s_1, ..., s_n$.\\

    TambiĂŠn puede haber varios nodos objetivo $\gamma_1, ..., \gamma_m\in \Gamma$, sin embargo, en este caso no se puede simplificar a un Ăşnico estado final.\\

    El problema por tanto serĂĄ ir de un Ăşnico estado inicial $s$ a alguno de los estados finales $\gamma\in \Gamma$.
\end{observacion}

\section{Algoritmo $A^*$}
Este algoritmo utiliza como funciĂłn de estimaciĂłn $f(n)=g(n)+h(n)$ para todo nodo $n$. Por la naturaleza del problema se verificarĂĄ que $g(s)=0$ y $h(\gamma)=0$. AdemĂĄs, se verifica que 
\begin{gather*}
    f^*(n) = g^*(n) + h^*(n)
\end{gather*}
Donde se verificarĂĄ tambiĂŠn
\begin{gather*}
    f^*(s)=h^*(s) = g^*(\gamma) = f^*(\gamma) = c^* \ \ \ \ \forall \gamma \in \Gamma
\end{gather*}
En general, por la optimalidad se tendrĂĄ que, para cualquier camino $P$ se tiene que
\begin{gather*}
    g_P(n)\geq g^*(n)\\
    h_P(n) \geq h^*(n)
\end{gather*}

\begin{propiedades}
    Queremos estudiar ahora 2 propiedades del algoritmo $A^*$:
    \begin{enumerate}
        \item $f^*(n) = c^*$ para todo $n\in P^*_{s-\Gamma}$.
        \begin{proof}
            Sea $n\in P^*_{s-\Gamma}$, entonces tenemos que existe un camino $P$ tal que
            \begin{gather*}
                g_P(n) + h_P(n) = c^*
            \end{gather*}
            Por la optimalidad se tendrĂĄ que 
            \begin{gather*}
                g^*(n) + h^*(n) \leq c^*
            \end{gather*}
            y necesariamente se tendrĂĄ que dar la igualdad, ya que en caso contrario $c^*$ no serĂ­a Ăłptimo y llegarĂ­amos a contradicciĂłn.
        \end{proof}
        \item $f^*(n) > c^*$ para todo $n\notin P^*_{s-\Gamma}$.
        \begin{proof}
            Supongamos que $f^*(n)\leq c^*$. Entonces tendremos que 
            \begin{gather*}
                g^*(n) + h^*(n) \leq c^*
            \end{gather*}
            luego existe un camino $P'$ tal que $g_P(n) + h_P(n) \leq c^*$. En caso de que se de la igualdad tendremos que $P'$ serĂ­a Ăłptimo y tendrĂ­amos que $n\in P^*_{s-\Gamma}$. En caso de ser menor tendrĂ­amos que $c^*$ no es el coste mĂ­nimo, lo que tambiĂŠn nos llevarĂ­a a contradicciĂłn.
        \end{proof}
    \end{enumerate}
\end{propiedades}

\section{HeurĂ­sticas monĂłtonas}

Por la desigualdad triangular tenemos que $k(s,\gamma) \leq k(s,n) + k(n, \gamma)$ para cualquier nodo $n$, cualquier nodo inicial $s$ y final $\gamma$. Tenemos por tanto que 
\begin{gather*}
    h^*(s) \leq k(s,n) + h^*(n) \Rightarrow h^*(n) \leq k(n,n') + h^*(n') \ \ \ \ \forall n,n' \ \text{ nodos}
\end{gather*}
\begin{definicion}
    Diremos que $h$ es \textbf{consistente} si se verifica que 
    \begin{gather*}
        h^*(n) \leq k(n,n') + h^*(n') \ \ \ \ \forall n,n' \ \text{ nodos}
    \end{gather*}
\end{definicion}

\begin{definicion}
    % Si consideramos la desigualdad anterior entre un nodo padre y sus hijos llegamos a la siguiente expresiĂłn:
    Diremos que $h$ es \textbf{monĂłtona} si
    \begin{gather*}
        h(n)\leq c(n,n')+h(u) \ \ \ \ \forall n,\ \ \forall n'\in suc(u)
    \end{gather*}
\end{definicion}

\begin{prop}
    Una heurĂ­stica $h$ es consistente si y solo si es monĂłtona.
\end{prop}

\begin{teo}
    Toda heurĂ­stica consistente es admisible
    \begin{proof}
        Por ser $h$ consistente se tiene que verificar la desigualdad
        \begin{gather*}
            h^*(n) \leq k(n,n') + h^*(n') \ \ \ \ \forall n,n' \ \text{ nodos}
        \end{gather*}
        Si tomamos $n'=\gamma \in \Gamma^*$ se tiene que 
        \begin{gather*}
            h(n) \leq k(n,\gamma) + h(\gamma) \Rightarrow h(n) \leq h^*(n) + 0 = h^*(n)
        \end{gather*}
        para cualquier nodo $n$ lo cual nos dice que $h$ es admisible.
    \end{proof}
\end{teo}

\begin{teo}
    Un algoritmo $A^*$ guiado por una heurĂ­stica monĂłtona alcanza el camino Ăłptimo a todos los nodos expandidos, es decir, 
    \begin{gather*}
        g(n) = g^*(n)\ \ \ \ \forall n \in \text{CERRADOS}
    \end{gather*}
    \begin{proof}
        $A^*$ selecciona como mejor nodo $n$. Supongamos que $g(n)>g^*(n)$, entonces 
        \begin{equation*}
            \text{ABIERTOS} = \{... n ...\}
        \end{equation*}
        Por el lema 1 tenemos que $\exists n'\in$ABIERTOS tal que $u'\in P^*_{s-n}$, en cuyo caso, $g(n')=g^*(n')$. Tenemos entonces
        \begin{gather*}
            f(n) = g(n') + h(n') \overset{\text{lema 1}}{=} g^*(n') + h(n') \overset{consist.}{\leq} g^*(n') + k(n',n) + h(n) \\
            \overset{opt.}{=} g^*(u)+h(n) \overset{hip.}{<} g(n)+h(n)=f(n)
        \end{gather*}
        y llegamos a contradicciĂłn.
    \end{proof}
\end{teo}
