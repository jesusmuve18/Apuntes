<tipo> & <identificador> = <iniciador>


Ejemplo 1:

int a=0; //Necesario para evitar el error de compilación (iniciador)
int &ref=a;
ref=5; //ahora "a" vale 5
cout<<a<<endl;


Ejemplo 2:

int v[5]={1,2,3,4,5};
int &ref=v[3]; //Modifica el dato de memoria almacenado en la posición 3 del array
ref=0;
cout<<v[3]<<endl;  // ahora vale 0
